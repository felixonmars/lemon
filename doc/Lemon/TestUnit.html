def description
  if meta?
    #"#{testcase} .#{target} #{aspect}"
    "#{testcase}.#{target} #{context} #{aspect}".strip
  else
    a  = /^[aeiou]/ =~ testcase.to_s ? 'An' : 'A'
    #"#{a} #{testcase} receiving ##{target} #{aspect}"
    "#{testcase}##{target} #{context} #{aspect}".strip
  end
end

The file_and_line method returns the file name and line number of the caller created upon initialization of this object.

This method is cached.

Examples

file_and_line #=> ['foo_test.rb', 123]

Returns Array of file name and line number of caller.

# File lib/lemon/model/test_unit.rb, line 128
def file_and_line
  @file_and_line ||= (
    line = caller[0]
    i = line.rindex(':in')
    line = i ? line[0...i] : line
    f, l = File.basename(line).split(':')
    [f, l]
  )
end
