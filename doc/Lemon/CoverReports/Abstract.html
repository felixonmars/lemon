def tally
  c = covered_units.size
  u = uncovered_units.size
  t = c + u

  pc = c * 100 / t
  pu = u * 100 / t

  "#{pc}% #{c}/#{t} covered, #{pu}% #{u}/#{t} uncovered" + 
  " (#{undefined_units.size} undefined units, #{uncovered_cases.size} uncovered cases)"
end
