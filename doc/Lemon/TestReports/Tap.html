def error(unit, exception)
  puts "not ok #{@i} - #{unit.description}"
  puts "  ERROR #{exception.class}"
  puts "  #{exception}"
  puts "  " + exception.backtrace.join("\n        ")
end

def start_suite(suite)
  @start = Time.now
  @i = 0
  n = suite.testcases.inject(0){ |c, tc| c = c + tc.size; c }
  puts "1..#{n}"
end
