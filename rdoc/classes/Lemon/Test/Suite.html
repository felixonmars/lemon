    def loadfiles(*files)
      Lemon.suite = self

      # directories glob *.rb files
      files = files.flatten.map do |file|
        if File.directory?(file)
          Dir[File.join(file, '**', '*.rb')]
        else
          file
        end
      end.flatten.uniq

      files.each do |file|
        #file = File.expand_path(file)
        #instance_eval(File.read(file), file)
        load(file)
      end

      return Lemon.suite
    end
